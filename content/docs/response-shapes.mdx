---
title: Response Shapes
description: TypeScript shapes for the currently implemented public API
---

## PostSummary

```ts
type PostSummary = {
  id: string;
  title: string;
  slug: string;
  excerpt: string;
  readingTimeMinutes: number;
  publishedAt: string | null;
  updatedAt: string;
  author: { id: string; name: string } | null;
  category: { slug: string; name: string } | null;
  tags: Array<{ slug: string; name: string }>;
};
```

## PostDetail

```ts
type PostDetail = PostSummary & {
  contentHtml: string;
  contentJson: unknown;
};
```

## WorkspaceMeta

```ts
type WorkspaceMeta = {
  id: string;
  name: string;
  slug: string;
};
```

## PostListResponse

```ts
type PostListResponse = {
  ok: true;
  workspace: WorkspaceMeta;
  posts: PostSummary[];
};
```

## PostDetailResponse

```ts
type PostDetailResponse = {
  ok: true;
  workspace: WorkspaceMeta;
  post: PostDetail;
};
```

## ErrorResponse

```ts
type ErrorResponse = {
  ok: false;
  error: string;
};
```

## Author

```ts
type Author = {
  id: string;
  name: string;
  email: string;
  about: string;
  socialLinks: Record<string, string>;
};
```

## Category

```ts
type Category = {
  id: string;
  name: string;
  slug: string;
};
```

## Tag

```ts
type Tag = {
  id: string;
  name: string;
  slug: string;
  createdAt: string;
};
```

## Stats

```ts
type Stats = {
  totalPosts: number;
  totalAuthors: number;
  totalCategories: number;
  totalTags: number;
};
```

## AuthorListResponse

```ts
type AuthorListResponse = {
  ok: true;
  workspace: WorkspaceMeta;
  authors: Author[];
};
```

## CategoryListResponse

```ts
type CategoryListResponse = {
  ok: true;
  workspace: WorkspaceMeta;
  categories: Category[];
};
```

## TagListResponse

```ts
type TagListResponse = {
  ok: true;
  workspace: WorkspaceMeta;
  tags: Tag[];
};
```

## StatsResponse

```ts
type StatsResponse = {
  ok: true;
  workspace: WorkspaceMeta;
  stats: Stats;
};
```

## Notes

- Current error responses use `error`, not `message`, for route-level failures.
- These shapes match the endpoints currently implemented in `v1`.
- `readingTimeMinutes` is an estimated read duration that accounts for text, images, and code blocks.
